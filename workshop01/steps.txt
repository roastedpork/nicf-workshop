# What I did
0. git clone https://github.com/stackupiss/cfdasa.git
1. cd into cfsasa/northwind
2. create docker-compose.yml in ./
3. create Dockerfile in  ./server
4. docker-compose build
5. docker-compose up -d

# What you wanted
0. git clone https://github.com/stackupiss/cfdasa.git
1. cd into cfsasa/northwind
2. docker image pull stackupiss/northwind-db:v1
3. docker network create mynet --driver bridge
4. docker volume create db-data
5. create Dockerfile in ./server
6. docker build -t myapp:latest ./server
7. docker run -d --name db --network mynet -v db-data:/var/lib/mysql -t stackupiss/northwind-db:v1
8. docker run -d --name app --network mynet -p 80:3000 -t myapp:latest
